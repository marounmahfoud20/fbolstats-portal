			});

$(function() {
	$('.slider').on('click', function() {

			$('#switch').attr('data-gender', 'female')
		} else {
			$('#switch').attr('data-gender', 'male')
		}

	});

	$('tr').each(function(i) {
		var tr = $(this);
		$(this).on('change','.teamedit_start_day, .teamedit_start_month, .teamedit_start_year, .teamedit_end_year, .teamedit_end_month, .teamedit_end_day',function() {
			var tpid = tr.find('input[type=hidden]').val();
			var first_mdate;
			var last_mdate;
			var startdate = tr.find('.teamedit_start_year').val() + "-" + tr.find('.teamedit_start_month').val() + "-" + tr.find('.teamedit_start_day').val();
			var enddate = tr.find('.teamedit_end_year').val() + "-" + tr.find('.teamedit_end_month').val() + "-" + tr.find('.teamedit_end_day').val();
			var main = $(this);
			
			$.ajax({
				url:'team_people_check_ajax.php',
				dataType: 'json',
				type: "POST",
				data: {tpid:tpid},
				success: function(data) {
					
					first_mdate = data.startdate;
					last_mdate = data.enddate;
						
					if (startdate > first_mdate) {
						tr.find('.teamedit_start_year').css("background-color","#ffcccc");
						tr.find('.teamedit_start_month').css("background-color","#ffcccc");
						tr.find('.teamedit_start_day').css("background-color","#ffcccc");
					}
					else {
						tr.find('.teamedit_start_year').css("background-color","white");
						tr.find('.teamedit_start_month').css("background-color","white");
						tr.find('.teamedit_start_day').css("background-color","white");								
					}
					
					if (enddate != '--' && enddate < last_mdate) {
						tr.find('.teamedit_end_year').css("background-color","#ffcccc");
						tr.find('.teamedit_end_month').css("background-color","#ffcccc");
						tr.find('.teamedit_end_day').css("background-color","#ffcccc");
					}
					else {
						tr.find('.teamedit_end_year').css("background-color","white");
						tr.find('.teamedit_end_month').css("background-color","white");
						tr.find('.teamedit_end_day').css("background-color","white");
					}
				}										
			});
		});
	});
});

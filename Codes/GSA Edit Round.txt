		var sport = $(this).data('id');

			url: 'sess.php',

			}
		});
	});
	
	$('.nav-search input').on('input', function() {
		var val = $(this).val().toLowerCase();
		var items = $(this).parent().parent().find('.nav-item-wrapper > *');
		
		if (val != '') {
			items.hide();
			
			items.each(function(el) {
				if ($(this).text().toLowerCase().indexOf(val) >= 0) {
					$(this).show();
				}
			});
		} else {
			items.show();
		}
	});

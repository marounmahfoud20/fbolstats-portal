$(function() {
	$('body').addClass('collapsed');
	$('.nav-toggle').on('click', function() {
		if ($('#main-navigation').hasClass('expanded')) {
			$('#main-navigation').toggleClass('expanded');
			$('body').toggleClass('expanded');
			document.cookie = "leftBar=collapsed; path=/; expires=Fri, 31 Dec 9999 23:59:59 GMT";
		} else {
			$('#main-navigation').toggleClass('expanded');
			$('body').toggleClass('expanded');
			document.cookie = "leftBar=expanded; path=/; expires=Fri, 31 Dec 9999 23:59:59 GMT";
		}
	});
	$('.nav-comp-toggle').on('click', function() {
		$('.nav-sports').hide();
		$('.nav-comps').animate({width: 'toggle'});
	});
	$('.nav-sport-toggle').on('click', function() {
		$('.nav-comps').hide();
		$('.nav-sports').animate({width: 'toggle'});
	});
	$('.nav-sport').on('click', function() {
		var sport = $(this).data('id');
		$.ajax({
			url: 'sess.php',
			type: "POST",
			data: {sport_s: sport},
			success: function() {
				window.location.reload(true);
			}
		});
	});
	
	$('.nav-search input').on('input', function() {
		var val = $(this).val().toLowerCase();
		var items = $(this).parent().parent().find('.nav-item-wrapper > *');
		
		if (val != '') {
			items.hide();
			
			items.each(function(el) {
				if ($(this).text().toLowerCase().indexOf(val) >= 0) {
					$(this).show();
				}
			});
		} else {
			items.show();
		}
	});
});

</script>

$(function() {
	$('.slider').on('click', function() {
		
		var type = $(this).data('type');
		var target = '#' + type;
		var type_id = 0;
		var _this = $(this);
		if ($(this).hasClass('on')) {
			if (type == 'closing_round') { type_id = 2; }
			else if (type == 'closing_season') { type_id = 4; }
			else if (type == 'closing_comp') { type_id = 6; }
			else if (type == 'season_over_check') { type_id = 8; }
		}
		else {
			if (type == 'closing_round') { type_id = 1; }
			else if (type == 'closing_season') { type_id = 3; }
			else if (type == 'closing_comp') { type_id = 5; }
			else if (type == 'season_over_check') { type_id = 7; }
		}
		
		$.ajax({
			url: './widget/closing_checks.php',
			type: "POST",
			dataType: 'json',
			data: {type_id: type_id, user_view: 101, session_user: 'maroun'},
			success: function(data) {
				
				if (_this.hasClass('on')) {
					_this.removeClass('on');
				}
				else {
					_this.addClass('on');
				}
				
				
				$(target).html(data.result);
			}
		});
		
		
	});
});

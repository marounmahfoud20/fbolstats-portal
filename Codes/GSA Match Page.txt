		if (ast_val == 'soc_G' || ast_val == 'flb_G' || ast_val == 'flb_SHG' || ast_val == 'flb_PG' || ast_val == 'ice_PG' || ast_val == 'ice_G' || ast_val == 'ice_SHG' || ast_val == 'ice_ENG' || ast_val == 'flb_DPS' || ast_val == 'flb_SHEN' || ast_val == 'flb_ENG' || ast_val == 'flb_WG' || ast_val == 'flb_ESH') {
			$('.ast_event_cont').show();
		} else if (ast_val == 'rug_SI') {
			$('.rug_sub_event').show();
		}
		else {
			$('.rug_sub_event').hide();
			$('.ast_event_cont').hide();
		}
	});
	
	$('.eventsadding2').on('click', '#add_event_save_match_event input[name=event]', function() {
		var ast_val = $(this).val();
		var extra = $(this).data('extra');
		var rel = $(this).data('rel');
		var rel2 = $(this).data('rel2');
		
		if (rel) {
			$('.rel_event_1').show();
		} else {
			$('.rel_event_1').hide();
		}
		
		if (extra == 'booking_reason') {
			$('select[name=booking_reason]').show();
			$('select[name=goal_type]').hide();
		} else if (extra == 'goal_type') {
			$('select[name=booking_reason]').hide();
			$('select[name=goal_type]').show();
		} else {
			$('select[name=booking_reason]').hide();
			$('select[name=goal_type]').hide();
		}
		
		if ($("select[data-type]").length) {		
			$("select[data-type]").hide();
		}
		
		if ($("select[data-extra-type]").length) {
			$("select[data-extra-type]").hide();
		}
		
		if ($("select[data-type='"+ast_val+"']").length) {		
			$("select[data-type='"+ast_val+"']").show();
		} else if ($("[data-extra-type='"+extra+"']").length) {
			$("[data-extra-type='"+extra+"']").show();
		}
	});

		var match = $(this).data('match');
		var people = $(this).data('people');

		$.ajax({
			url:'injury_match_delete.php',
			dataType: 'json',
			type: "POST",
			data: {match_id: match, people_id: people},
			success: function(data){
				$('#injuries_container').empty();
				$('#injuries_container').html(data.result);
			}			
		});
	});

	$('.lol_icons,.lol_icons_b').on('click', function() {		
		var type = $(this).attr('title');
		var incr;
		if (type == 'gold' || type == 'gold_b') { incr = 0.1; } else { incr = 1; }		
		var val_temp = $(this).parent().find('.'+type).val();	
		var val = parseFloat(val_temp) + incr;
		var period = $(this).parent().parent().find('.lol_period').val();		
		var target = $(this).parent().find('.'+type);
		
		if (val_temp == '') { val = 0; }
		
		$.ajax({
			url: 'lol_stat_widget.php',
			dataType: 'json',
			type: "POST",
			data: {match_id: 3902314, type: type, val: val, period: period},
			success: function(data) {
				if (data.result) {				
					target.val(val);
				}
			}
		}); 
		
	});

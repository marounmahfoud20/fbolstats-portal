		$('.nav-comps').hide();
		$('.nav-sports').animate({width: 'toggle'});
	});

		var sport = $(this).data('id');
		$.ajax({
			url: 'sess.php',
			type: "POST",
			data: {sport_s: sport},
			success: function() {
				window.location.reload(true);
			}
		});
	});
	
	$('.nav-search input').on('input', function() {
		var val = $(this).val().toLowerCase();
		var items = $(this).parent().parent().find('.nav-item-wrapper > *');
		
		if (val != '') {
			items.hide();
			
			items.each(function(el) {
				if ($(this).text().toLowerCase().indexOf(val) >= 0) {
					$(this).show();
				}
			});
		} else {
			items.show();
		}
	});
});

		if (target.css('display') == 'none') {
			$.ajax({
				url: 'https://dsg-backend.com/internal/include/people_view/matches_ajax.php',
				type: 'post',
				dataType: 'json',
				data: {type: type, id: id},

					target.html(data.result);

				}
			});
		} else {
			target.empty();
			target.hide();
		}
		
		
	});

			}
		});

		e.preventDefault();

			$(this).children('img.off').css('display','inline');
			$(this).children('img.on').css('display','none');
			$('#match_container').empty();
			$('#match_container').css('display','block');
			$('#load_gif').css('display','block');
			
			$.ajax({
				url: "p_matches_ajax.php",
				dataType: "json",
				type: "POST",
				data: {pid: 44318, sport: "soccer"},
				success: function(data) {
					$('#load_gif').css('display','none');
					$('#match_container').html(data.result);
				}
			});
		}
		else {
			$('#match_container').css('display','none');
			$('#datalog_load_gif').css('display','none');
			$(this).children('img.on').css('display','inline');
			$(this).children('img.off').css('display','none');
		}

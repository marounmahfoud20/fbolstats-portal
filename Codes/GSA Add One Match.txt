$(function() {
	$('body').addClass('collapsed');
	$('.nav-toggle').on('click', function() {
		if ($('#main-navigation').hasClass('expanded')) {
			$('#main-navigation').toggleClass('expanded');
			$('body').toggleClass('expanded');
			document.cookie = "leftBar=collapsed; path=/; expires=Fri, 31 Dec 9999 23:59:59 GMT";
		} else {
			$('#main-navigation').toggleClass('expanded');
			$('body').toggleClass('expanded');
			document.cookie = "leftBar=expanded; path=/; expires=Fri, 31 Dec 9999 23:59:59 GMT";
		}
	});
	$('.nav-comp-toggle').on('click', function() {
		$('.nav-sports').hide();
		$('.nav-comps').animate({width: 'toggle'});
	});
	$('.nav-sport-toggle').on('click', function() {
		$('.nav-comps').hide();
		$('.nav-sports').animate({width: 'toggle'});
	});
	$('.nav-sport').on('click', function() {
		var sport = $(this).data('id');
		$.ajax({
			url: 'sess.php',
			type: "POST",
			data: {sport_s: sport},
			success: function() {
				window.location.reload(true);
			}
		});
	});
	
	$('.nav-search input').on('input', function() {
		var val = $(this).val().toLowerCase();
		var items = $(this).parent().parent().find('.nav-item-wrapper > *');
		
		if (val != '') {
			items.hide();
			
			items.each(function(el) {
				if ($(this).text().toLowerCase().indexOf(val) >= 0) {
					$(this).show();
				}
			});
		} else {
			items.show();
		}
	});
});

</script>
